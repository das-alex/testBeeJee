create table users (
  id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_name varchar(30),
  password varchar(30),
  email varchar(30),
  isAdmin int(1)
);

create table tasks (
  id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id int(6),
  task varchar(200),
  status int(1)
);
